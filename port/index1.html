//영화차트 탭 메뉴
var movBtn = $(".movie_title > ul > li");
var movCont = $(".movie_chart > div");
movCont.hide().eq(0).show();       //첫번째만 보이고 나머지 내용들 안보이게 숨기기
    
movBtn.click(function(e){
    e.preventDefault();            // 버튼을 클릭시 상단으로 올라가는 기능 없애기
    var target = $(this);          //movBtn = this
    var index = target.index();
    movBtn.removeClass("active");  //기존에 있는 엑티브 클레스를 없애주기
    target.addClass("active");
    movCont.css("display","none"); //기존의 블록화 되어있는 내용들을 숨김처리 해놓기
    movCont.eq(index).css("display","block"); 
});
